'id' = Identifier
'totalCost'= total cost of the project
'ecMaxContribution' = contribution of the project 
'pry_duration_m' = duration of the project in months
'pry_duration_d1' = duration of the project in days
'delay_m' = delay of the sign of the project in months
'delay_d1' = delay of the sign of the project in days
'ratio' = ratio between max contribution and total cost of the project
'title_y' = general topic of the project (has the same levels as "legalbasis")
'num_org' = Number of organizations involved in the project
'cost0' = indicates the projects with total cost = 0
'country1' = country where the coordinator is from
